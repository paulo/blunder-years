%{
#include "musica.h"

char* takeOffAnotations(char* c){
    int i=0;
    while(c[i] != '\0' && !(c[i] == ' ' && c[i + 1] == ' ')) {
        i++;
    }
    c[i] = '\0';
    return c;
}

char* takeOffUnderSccore(char* c){
    int i=0, w=0;
    while(c[i] != '\0') {
        //ignore if '_'
        if(c[i] != '_') {
            c[w] = c[i];
            w++;
        }
        i++;
    }
    c[w] = '\0';
    return c;
}
